 39:     /**
 40:      * Get the current user of the application.
 41:      *
 42:      * If the user is a guest, null should be returned.
 43:      *
 44:      * @param  int         $id
 45:      * @return mixed|null
 46:      */
 47:     public function retrieve($id)
 48:     {
 49:         return $this->user()->make($identifier);
 50:     }

 52:     /**
 53:      * Access an instance of the User Model
 54:      *
 55:      * If no identifier is specified, should return an empty instance of the model.
 56:      *
 57:      * @return mixed|null
 58:      */
 59:     public function user($identifier = '')
 60:     {
 61:         // initialize $lookup. If all else fails we will use this.
 62:         $lookup = '';
 63: 
 64:         if ( ! empty($identifier))
 65:         {
 66:             $lookup = $identifier;
 67:         }
 68:         elseif ( ! empty($this->token))
 69:         {
 70:             $lookup = $this->token;
 71:         }
 72: 
 73:         return \SeptuUser::make($lookup);       
 74:     }

 77:      * Get instance of the Group Model
 78:      *
 79:      * @return mixed|null

 81:     public function group($identifier = '')
 82:     {
 83:         return \SeptuGroup::make($identifier);      
 84:     }

 87:      * Get instance of the Permissions Model
 88:      *
 89:      * @return mixed|null

 91:     public function permissions($identifier = '')

 93:         return \SeptuPermission::make($identifier);     
 94:     }

 96:     /**
 97:      * Determine if the user is logged in.
 98:      *
 99:      * @return bool
100:      */
101:     public function check()
102:     {
103:         // grab the user id from the token
104:         $user_id = $this->token;
105: 
106:         // if no user id, then check to see if we have a cookie
107:         if ($user_id === null or ! is_numeric($user_id))
108:         {
109:             $user_id = $this->recall();
110:         }
111: 
112:         // simple check to make sure the user is in the system
113:         $query = $this->user()->where('id', '=', $user_id);
114: 
115:         return $query->count() != 0;
116:     }

118:     /**
119:      * Login the user assigned to the given token.
120:      *
121:      * The token is typically a numeric ID for the user.
122:      *
123:      * @param  string  $token
124:      * @param  bool    $remember
125:      * @return bool
126:      */
127:     public function login($token, $remember = false)
128:     {
129:         $this->token = $token;
130: 
131:         $this->store($token);
132: 
133:         if ($remember) $this->remember($token);
134: 
135:         if (Config::get('septu::septu.use.logins'))
136:         {
137:             $login_attributes = array(
138:                 'ip_address' => Request::ip(),
139:             );
140: 
141:             $login_insert = new \SeptuProfile($login_attributes);
142:             $this->user()->logins()->insert($login_insert);
143:         }
144:         else
145:         {
146:             die('no login recording');
147:         }
148: 
149:         // update the user's login information
150:         $update = array(
151:             'last_login' => time(),
152:             'ip_address' => Request::ip(),
153:         );
154: 
155:         if ($this->user()->update_user($update))
156:         {
157:             return true;            
158:         }
159:     }

161:     /**
162:      * Attempt to log a user into the application.
163:      *
164:      * @param  array  $arguments
165:      * @return void
166:      */
167:     public function attempt($arguments = array())
168:     {
169:         // logout of any previous logins
170:         self::logout();
171: 
172:         // get the username field from the config. set to 'username' by default
173:         $username_field = Config::get('septu::septu.username.field', 'username');
174: 
175:         // try to find the user
176:         $user = $this->user($arguments['username']);
177: 
178:         // This driver uses a basic username and password authentication scheme
179:         // so if the credentials match what is in the database we will just
180:         // log the user into the application and remember them if asked.
181:         $password = $arguments['password'];
182: 
183:         // check to make sure we were able to find a matching user account
184:         if ( is_null($user))
185:         {
186:             throw new SeptuException(__('septu::septu.attempt_invalid_user'));
187:         }
188: 
189:         // validate the user's password
190:         $password_field = Config::get('septu::septu.password.field', 'password');
191: 
192:         if ( !  $this->_hasher()->check($password, $user->get_attribute($password_field)))
193:         {
194:             throw new SeptuException(__('septu::septu.attempt_bad_password'));
195:         }
196: 
197:         // make sure the user's accout has been enabled
198:         if ($user->activated == false)
199:         {
200:             throw new SeptuException(__('septu::septu.account_not_activated'));
201:         }
202: 
203:         // make sure the user's accout has been enabled
204:         if ($user->enabled == false)
205:         {
206:             throw new SeptuException(__('septu::septu.account_disabled'));
207:         }
208: 
209:         return $this->login($user->id, array_get($arguments, 'remember'));
210:     }
